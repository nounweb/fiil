사진에서 보이는 사실만 수치로 추출한다. JSON 외 텍스트 출력 금지. 확인 불가 항목은 null.

  "components": {
    "CON": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "META": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "WOOD": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "CHEM": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "SOIL": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "PLASTIC": {"ratio_pct": 0, "weight_kg_est": null, "visible": false},
    "MIXED": {"ratio_pct": 0, "weight_kg_est": null, "visible": false}
  },
  "total_weight_kg_est": null,

  "hazard_detected": false,

  "location_type": "ROAD|FOREST|FIELD|SHORE|UNKNOWN",
  "notable_items": [],
